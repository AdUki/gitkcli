# Detached HEAD (a commit checked out directly, not a branch tip): the log still
# renders, the detached commit carries the (HEAD) marker with NO "-> branch"
# arrow (head_branch is empty when detached), and there is no error. Build a
# tiny repo with fully-pinned dates+offsets so the commit hashes are
# deterministic, then detach at the root commit.
size      120x40
config    default
run       rm -rf .git && git init -q && git config user.email a@b.c && git config user.name Tester && GIT_AUTHOR_DATE='2024-01-01T00:00:00+0000' GIT_COMMITTER_DATE='2024-01-01T00:00:00+0000' git commit -q --allow-empty -m "root commit" && GIT_AUTHOR_DATE='2024-01-02T00:00:00+0000' GIT_COMMITTER_DATE='2024-01-02T00:00:00+0000' git commit -q --allow-empty -m "second commit" && git checkout -q HEAD~1
launch
capture   detached_head
