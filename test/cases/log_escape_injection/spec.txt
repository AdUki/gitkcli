# A commit subject containing raw ANSI escape sequences must NOT be passed
# through to the terminal (terminal-injection guard). The escapes are stripped
# in Job._reader_thread, leaving inert text. Build the subject with printf so the
# spec file holds no raw ESC bytes; pinned dates keep the hash deterministic.
size      120x40
config    default
run       rm -rf .git && git init -q && GIT_AUTHOR_DATE='2024-01-01T00:00:00+0000' GIT_COMMITTER_DATE='2024-01-01T00:00:00+0000' git commit -q --allow-empty -m "$(printf 'pre\033[31mRED\033[0m\033[2Apost')"
launch
capture   escape_injection
