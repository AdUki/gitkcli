# F5 refresh in --graph mode. The lane art is computed by git over the whole
# commit set, so a single new commit cannot be appended incrementally - the view
# must reload the full log. Launch with --graph, create an external commit on
# HEAD, press F5, then 'g' to scroll to the (reloaded) top where the new commit
# is visible with correct graph art.
size      120x40
config    default
launch    --graph
wait      stable
run       git commit --allow-empty -m "external graph commit"
key       <F5>
wait      2.5
key       g
wait      stable
capture   reloaded
