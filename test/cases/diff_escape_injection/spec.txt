# A file whose CONTENT contains control/escape bytes is stripped of them on the
# diff path too (same _reader_thread hygiene, DiffListItem rendering). curses
# would render control bytes as caret notation, not pass them through raw; this
# just keeps the display clean. printf keeps raw ESC out of the spec; pinned date
# -> deterministic hash.
size      120x40
config    default
run       rm -rf .git && git init -q && printf 'line one\npre\033[31mEVILDIFF\033[0m\033[2Jpost\nline three\n' > file.txt && git add file.txt && GIT_AUTHOR_DATE='2024-01-01T00:00:00+0000' GIT_COMMITTER_DATE='2024-01-01T00:00:00+0000' git commit -q -m "add file with escapes in content"
launch
key       <Enter>
capture   diff_escapes
